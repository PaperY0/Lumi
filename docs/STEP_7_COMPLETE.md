## 📝 技术细节

### 多行内容合并逻辑（关键算法）
```typescript
// 收集内容直到下一个"昵称+时间"组合
const contentLines: string[] = [];
let j = i + 2;  // 从时间行的下一行开始
while (j < lines.length) {
  // 检查 j 和 j+1 是否构成新的消息块（昵称+时间）
  if (
    lines[j] &&
    !TIME_REGEX.test(lines[j]) &&  // 当前行不是时间
    lines[j + 1] &&
    TIME_REGEX.test(lines[j + 1])   // 下一行是时间
  ) {
    break; // 遇到下一条消息
  }
  
  if (lines[j]) contentLines.push(lines[j]);
  j++;
}

const content = contentLines.join('\n').trim();
```

### 昵称行判定规则
```typescript
const nameLine = lines[i];
const timeLine = lines[i + 1];

// 当前行不是时间，且下一行是时间 → 当前行是昵称
if (!timeLine || !TIME_REGEX.test(timeLine)) {
  i++;
  continue;
}
```
